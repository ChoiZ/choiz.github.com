Git mes fichiers de config gitconfig / gitignore

Posté le mar. 31 janvier 2012

Fichier .gitconfig :

[user]
    name = François LASSERRE
    email = mon@email.fr
[diff]
    tool = vimdiff
[color]
    branch = auto
    diff = auto
    status = auto
    interactive = auto
    ui = true
[alias]
    br = branch
    ci = commit -a
    co = checkout
    sh = show --color-words
    st = status
    last = cat-file commit HEAD
    tree = log --graph --oneline --decorate
    graph = log --graph --decorate
    pom = push origin master
[core]
    excludesfile = ~/.gitignore
    editor = vim
[merge]
    tool = vimdiff
